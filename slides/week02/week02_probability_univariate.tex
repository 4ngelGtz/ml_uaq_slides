\begin{frame}{¿Por qué es relevante el enfoque Bayesiano?}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\begin{itemize}
  \item \key{Objetivo:} inferir \(\theta=\Pr(y=1)\) (una \key{probabilidad}) a partir de datos \(D\).
  \item \key{Incertidumbre explícita:} obtenemos una distribución \(p(\theta\mid D)\), no solo un punto.
  \item \key{Actualización coherente:} \(\text{prior}+\text{datos}\Rightarrow\) \text{posterior}.
  \item \key{Predicción robusta:} la \key{posterior predictiva} integra incertidumbre \(\Rightarrow\) menos sobreconfianza con pocos datos.
  \item \key{Regularización interpretable:} priors como \key{pseudoconteos} (suavizado).
  \item \key{Comparación de modelos/priors:} evidencia \(p(D)\) = \key{probabilidad marginal de los datos}.
\end{itemize}
\end{frame}

\begin{frame}{Enfoque frecuentista (ejemplo: tasa de compra)}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\begin{itemize}
  \item Datos: \(n=20\), \(k=3\) \(\Rightarrow\) \(\hat\theta=0.15\).
  \item Estimación: \(\hat\theta=k/n\) (MLE).
  \item IC 95\% (aprox. normal):
  \[
    \hat\theta \pm 1.96\sqrt{\frac{\hat\theta(1-\hat\theta)}{n}}
    =0.15\pm 0.156\Rightarrow [-0.006,\,0.306]
  \]
  \item Intervalo válido en \([0,1]\): \([0,\,0.306]\).
  \item Interpretación: 95\% de los ICs (en repeticiones) contienen a \(\theta\).
\end{itemize}
\end{frame}

\begin{frame}{Enfoque Bayesiano (mismo ejemplo)}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\begin{itemize}
  \item Prior: \(\theta\sim\mathrm{Beta}(2,8)\) (creencia previa \(\approx 0.20\)).
  \item Posterior: \(\theta\mid D\sim\mathrm{Beta}(5,25)\).
  \item Media posterior: \(\mathbb{E}[\theta\mid D]=\dfrac{5}{30}\approx 0.167\).
  \item Intervalo creíble 95\%: percentiles de \(\mathrm{Beta}(5,25)\).
  \item Interpretación: \(P(\theta\in \text{ICr}\mid D)=0.95\).
\end{itemize}
\end{frame}

\section[Teoría]{Teoría\label{sec:teoria}}

\begin{frame}{Motivación: inferir una probabilidad con pocos datos}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\begin{itemize}
  \item Problema base: estimar \(\theta=\Pr(y=1)\) a partir de pocos ensayos Bernoulli.
  \item Bayes da \key{distribución} sobre \(\theta\), no solo un número.
  \item Ventaja: incorpora información previa y cuantifica incertidumbre.
  \item Caso canónico: Bernoulli/Binomial + Beta \(\Rightarrow\) actualizaciones cerradas.
\end{itemize}
\end{frame}

\begin{frame}{Definiciones (2.3): piezas de Bayes}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\begin{itemize}
  \item \key{Prior:} \(p(\theta)\).
  \item \key{Likelihood:} \(p(D\mid\theta)\).
  \item \key{Posterior:} \(p(\theta\mid D)=\dfrac{p(D\mid\theta)\,p(\theta)}{p(D)}\).
  \item \key{Evidencia:} \(p(D)=\int p(D\mid\theta)\,p(\theta)\,d\theta\) (\term{marginal likelihood}).
\end{itemize}
\end{frame}

\begin{frame}{Derivación 1 (2.4): likelihood Bernoulli/Binomial}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\begin{itemize}
  \item Datos iid \(y_i\in\{0,1\}\), con \(k=\sum_{i=1}^n y_i\).
  \item Bernoulli: \(p(y_i\mid\theta)=\theta^{y_i}(1-\theta)^{1-y_i}\).
  \item Likelihood agregada:
  \[
    p(D\mid\theta)=\prod_{i=1}^n \theta^{y_i}(1-\theta)^{1-y_i}
    =\theta^{k}(1-\theta)^{n-k}.
  \]
  \item Binomial: \(p(k\mid n,\theta)=\binom{n}{k}\theta^{k}(1-\theta)^{n-k}\).
\end{itemize}
\end{frame}

\begin{frame}{Distribución Beta (2.7.4): prior sobre probabilidades}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\begin{itemize}
  \item Soporte: \(\theta\in(0,1)\) (ideal para probabilidades).
  \item Densidad:
  \[
    \theta\sim\mathrm{Beta}(\alpha,\beta),
    \qquad
    p(\theta)=\frac{1}{B(\alpha,\beta)}\theta^{\alpha-1}(1-\theta)^{\beta-1}.
  \]
  \item Lectura: \(\alpha,\beta\) controlan \key{sesgo} (media) y \key{fuerza} (concentración).
  \item Interpretación útil: \(\alpha,\beta\) se comportan como \key{pseudoconteos} de éxitos/fracasos.
\end{itemize}
\end{frame}

\begin{frame}{Priors Beta: ¿qué “forma” de creencia estás poniendo?}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\begin{columns}[c]
\column{0.58\textwidth}
\centering
\includegraphics[width=\textwidth]{slides/figs/week02_beta_priors.png}
\column{0.42\textwidth}
\begin{itemize}
  \item \(\mathrm{Beta}(1,1)\): uniforme.
  \item \(\mathrm{Beta}(2,2)\): suave, centrada en \(0.5\).
  \item \(\mathrm{Beta}(8,2)\): empuja hacia \(\theta\) altos.
  \vspace{0.4em}
  \item \key{Idea:} el prior dice qué valores de \(\theta\) son “plausibles” antes de ver datos.
\end{itemize}
\end{columns}
\end{frame}

\begin{frame}{Derivación 2 (4.6.1): conjugación Beta con Bernoulli/Binomial}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\begin{itemize}
  \item Usamos: \(p(\theta\mid D)\propto p(D\mid\theta)\,p(\theta)\).
  \item Como \(p(D\mid\theta)\propto \theta^{k}(1-\theta)^{n-k}\) y \(p(\theta)\propto \theta^{\alpha-1}(1-\theta)^{\beta-1}\),
  el posterior queda:
  \[
    p(\theta\mid D)\propto \theta^{k+\alpha-1}(1-\theta)^{n-k+\beta-1}.
  \]
  \item Conjugación (misma familia):
  \[
    \theta\mid D\sim \mathrm{Beta}(\alpha+k,\ \beta+n-k).
  \]
  \item Lectura rápida: \(k\) suma a “éxitos” y \(n-k\) suma a “fracasos”.
\end{itemize}
\end{frame}

\begin{frame}{¿Qué es la \key{posterior predictiva}?}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\begin{itemize}
  \item \key{Definición:} distribución de \key{datos futuros} dado lo observado:
  \[
    p(D_{\text{new}}\mid D)=\int p(D_{\text{new}}\mid \theta)\,p(\theta\mid D)\,d\theta.
  \]
  \item \key{Idea:} en vez de fijar \(\theta\) a un punto (plug-in), \key{promediamos} sobre la incertidumbre en \(\theta\).
  \item En Bernoulli/Binomial:
  \begin{itemize}
    \item 1 paso: \(p(y_{\text{new}}=1\mid D)=\mathbb{E}[\theta\mid D]\).
    \item \(M\) pasos: \(S=\sum_{j=1}^M y_j \mid D\) tiene una distribución (Beta--Binomial) más “ancha” que plug-in.
  \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Predictiva \term{plug-in} vs predictiva Bayesiana (en una línea)}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\begin{itemize}
  \item \key{Plug-in:} primero estimas \(\hat\theta\) (p.ej. MLE/MAP/media posterior) y luego predices como si fuera cierto:
  \[
    p_{\text{plug-in}}(D_{\text{new}}\mid D)=p(D_{\text{new}}\mid \hat\theta).
  \]
  \item \key{Bayes:} no fijas \(\theta\); promedias con el posterior:
  \[
    p(D_{\text{new}}\mid D)=\int p(D_{\text{new}}\mid \theta)\,p(\theta\mid D)\,d\theta.
  \]
  \item \key{Mensaje:} plug-in suele ser más “optimista”; Bayes tiende a ser menos sobreconfiante con pocos datos.
\end{itemize}
\end{frame}

\begin{frame}{Modelo Beta--Binomial (4.6.2): posterior y predictiva}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\begin{itemize}
  \item Predictiva 1 paso:
  \[
    p(y_{\text{nuevo}}=1\mid D)=\mathbb{E}[\theta\mid D]=\frac{\alpha+k}{\alpha+\beta+n}.
  \]
  \item Predictiva de conteos (\(m\) futuros):
  \[
    p(k_{\text{new}}\mid m,D)=\binom{m}{k_{\text{new}}}
    \frac{B(\alpha+k+k_{\text{new}},\beta+n-k+m-k_{\text{new}})}{B(\alpha+k,\beta+n-k)}.
  \]
\end{itemize}
\end{frame}

\begin{frame}{Ejemplo numérico mínimo}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\begin{itemize}
  \item Prior uniforme: \(\theta\sim \mathrm{Beta}(1,1)\).
  \item Datos: \(n=4\) lanzamientos, \(k=3\) éxitos.
  \item Posterior: \(\theta\mid D\sim \mathrm{Beta}(4,2)\).
  \item Predictiva: \(p(y_{\text{nuevo}}=1\mid D)=\dfrac{4}{6}=\dfrac{2}{3}\).
\end{itemize}
\end{frame}

\begin{frame}{Derivación (4.6.2.10): evidencia Beta--Binomial \(p(k\mid n,\alpha,\beta)\)}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\begin{itemize}
  \item Definición (marginalizar \(\theta\)):
  \[
    p(k\mid n,\alpha,\beta)=\int_0^1 p(k\mid n,\theta)\,p(\theta\mid\alpha,\beta)\,d\theta.
  \]
  \item Sustituimos Binomial y Beta:
  \[
    p(k\mid n,\theta)=\binom{n}{k}\theta^k(1-\theta)^{n-k},
    \quad
    p(\theta)=\frac{1}{B(\alpha,\beta)}\theta^{\alpha-1}(1-\theta)^{\beta-1}.
  \]
  \item Agrupando potencias:
  \[
    p(k\mid n,\alpha,\beta)=\binom{n}{k}\frac{1}{B(\alpha,\beta)}
    \int_0^1 \theta^{\alpha+k-1}(1-\theta)^{\beta+n-k-1}\,d\theta.
  \]
  \item Reconocemos la integral como \(B(\alpha+k,\beta+n-k)\) \(\Rightarrow\)
  \[
    p(k\mid n,\alpha,\beta)=\binom{n}{k}\,\frac{B(\alpha+k,\beta+n-k)}{B(\alpha,\beta)}.
  \]
\end{itemize}
\end{frame}

\begin{frame}{Evidencia \(p(D)\): intuición y comparación de modelos}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\begin{itemize}
  \item \key{Qué es:} \(p(D)\) es la \key{probabilidad marginal} del dataset (también \term{marginal likelihood}).
  \item \key{Rol en Bayes:} es el \key{normalizador} (no depende de \(\theta\)).
  \item \key{Intuición:} promedia el likelihood sobre \(\theta\) “plausibles” según el prior.
  \item \key{Comparación:} para dos modelos/priors \(M_1,M_2\),
  \[
    \text{Bayes factor}=\frac{p(D\mid M_1)}{p(D\mid M_2)}.
  \]
  \item \key{Lectura:} favorece modelos que explican bien los datos \key{sin} “apostar” probabilidad en demasiados \(\theta\).
\end{itemize}
\end{frame}

\begin{frame}{Evidencia: misma data, distinto prior \(\Rightarrow\) distinto \(p(D)\)}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\begin{itemize}
  \item Datos (Binomial): \(n=10\), \(k=7\) éxitos.
  \item Con prior \(\theta\sim\mathrm{Beta}(\alpha,\beta)\):
  \[
    p(k\mid n,\alpha,\beta)=\binom{n}{k}\,\frac{B(\alpha+k,\beta+n-k)}{B(\alpha,\beta)}.
  \]
  \item Evidencias (aprox.):
  \begin{itemize}
    \item \(\mathrm{Beta}(1,1)\): \(p(k\mid n)\approx 0.0909\)
    \item \(\mathrm{Beta}(2,2)\): \(p(k\mid n)\approx 0.1119\)
    \item \(\mathrm{Beta}(8,2)\): \(p(k\mid n)\approx 0.1486\)
  \end{itemize}
  \item \key{Interpretación (en simple):} estos números dicen qué tan “esperable” era ver \(7/10\) \emph{antes} de ver los datos, según cada prior.
  \item \key{Lo que importa: comparaciones.} Por ejemplo:
  \[
    \frac{p(D\mid \mathrm{Beta}(8,2))}{p(D\mid \mathrm{Beta}(1,1))}\approx 1.63,
    \qquad
    \frac{p(D\mid \mathrm{Beta}(2,2))}{p(D\mid \mathrm{Beta}(1,1))}\approx 1.23.
  \]
\end{itemize}
\end{frame}

\begin{frame}{Resumen y takeaways}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\begin{itemize}
  \item Bayes (2.3) combina prior y likelihood para obtener \(p(\theta\mid D)\).
  \item Bernoulli/Binomial (2.4) \(\Rightarrow\) likelihood \(\theta^{k}(1-\theta)^{n-k}\).
  \item Beta (2.7.4) \(\Rightarrow\) prior natural sobre probabilidades.
  \item Conjugación (4.6.1): \(\mathrm{Beta}(\alpha,\beta)\to \mathrm{Beta}(\alpha+k,\beta+n-k)\).
  \item Predictiva (4.6.2): promedia sobre \(\theta\) \(\Rightarrow\) menos sobreconfianza.
  \item Evidencia (4.6.2.10): \(p(D)\) sirve para \key{comparar} priors/modelos (Bayes factor).
\end{itemize}
\end{frame}

\section[Lab NumPy]{Lab NumPy\label{sec:labnumpy}}

\begin{frame}{Lab NumPy — objetivos (qué debes lograr)}
\temariobar{Teoría}{\textbf{Lab NumPy}}{Lab sklearn}{Entregable}

\begin{itemize}
  \item \key{Inferencia:} a partir de \(y\in\{0,1\}^n\), calcular \(N_1,N_0\) y actualizar
  \((\alpha,\beta)\rightarrow(\alpha',\beta')\).
  \item \key{Predicción:} calcular \(p(y_{\text{new}}=1\mid D)\) y aproximar \(p(S\mid D)\) para \(M\) futuros.
  \item \key{Comparación:} contrastar predictiva \term{plug-in} vs \term{Bayesiana} y explicar el efecto del prior
  (sesgo y fuerza) en 2--3 frases.
\end{itemize}
\end{frame}

\begin{frame}{Lab NumPy — definiciones mínimas (puente con Teoría)}
\temariobar{Teoría}{\textbf{Lab NumPy}}{Lab sklearn}{Entregable}
\begin{itemize}
  \item \key{Parámetro:} \(\theta=\Pr(y=1)\).
  \item \key{Datos:} \(D=\{y_i\}_{i=1}^n\), con conteos \(N_1=\sum_i y_i\), \(N_0=n-N_1\).
  \item \key{Prior:} \(\theta\sim\mathrm{Beta}(\alpha,\beta)\).
  \item \key{Posterior (conjugación):} \(\theta\mid D\sim\mathrm{Beta}(\alpha',\beta')\),
  donde \(\alpha'=\alpha+N_1\), \(\beta'=\beta+N_0\).
  \item \key{Posterior predictiva:} \(p(D_{\text{new}}\mid D)=\int p(D_{\text{new}}\mid\theta)p(\theta\mid D)d\theta\).
  \item \key{Plug-in vs Bayes:} plug-in fija \(\theta=\hat\theta\); Bayes integra sobre \(\theta\).
\end{itemize}
\end{frame}

\begin{frame}[fragile]{Lab NumPy — Pregunta 1: ¿Qué datos tengo y qué resumen necesito?}
\temariobar{Teoría}{\textbf{Lab NumPy}}{Lab sklearn}{Entregable}

\begin{verbatim}
ENTRADA: y[1..n]  (valores 0/1)

N1 <- suma(y)
N0 <- n - N1

SALIDA: N1, N0
CHEQUEO: N1 + N0 = n
\end{verbatim}
\begin{itemize}
  \item Definición: \(N_1\) = número de éxitos, \(N_0\) = número de fracasos.
\end{itemize}
\end{frame}

\begin{frame}[fragile]{Lab NumPy — Pregunta 2: ¿Qué creía antes de ver datos? (prior)}
\temariobar{Teoría}{\textbf{Lab NumPy}}{Lab sklearn}{Entregable}

\begin{verbatim}
ELIGE el prior (alpha > 0, beta > 0)

EJEMPLOS:
  alpha = 1, beta = 1   (uniforme)
  alpha = 2, beta = 2   (suave, centrado en 0.5)
  alpha = 8, beta = 2   (sesgo hacia theta altos)
\end{verbatim}
\begin{itemize}
  \item Tip: \(\alpha+\beta\) controla la “fuerza” del prior.
  \item Pitfall: prior fuerte sin justificar.
\end{itemize}
\end{frame}

\begin{frame}[fragile]{Lab NumPy — Pregunta 3: ¿Qué creo después de ver datos? (posterior)}
\temariobar{Teoría}{\textbf{Lab NumPy}}{Lab sklearn}{Entregable}

\begin{verbatim}
ACTUALIZA (conjugacion Beta--Bernoulli):
alpha_post <- alpha + N1
beta_post  <- beta  + N0

SALIDA: alpha_post, beta_post
CHEQUEO: alpha_post > 0 y beta_post > 0
\end{verbatim}
\begin{itemize}
  \item Definición: \(\alpha'=\alpha+N_1\), \(\beta'=\beta+N_0\).
\end{itemize}
\end{frame}

\begin{frame}[fragile]{Lab NumPy — Pregunta 4: ¿Qué predigo del siguiente ensayo? (predictiva 1 paso)}
\temariobar{Teoría}{\textbf{Lab NumPy}}{Lab sklearn}{Entregable}

\begin{verbatim}
PREDICTIVA POSTERIOR (1 ensayo):
p1_bayes <- alpha_post / (alpha_post + beta_post)
p0_bayes <- 1 - p1_bayes

SALIDA: p(y_nuevo = 1 | D) = p1_bayes
\end{verbatim}
\begin{itemize}
  \item Definición: \(p(y_{\text{new}}=1\mid D)=\mathbb{E}[\theta\mid D]=\dfrac{\alpha'}{\alpha'+\beta'}\).
\end{itemize}
\end{frame}

\begin{frame}[fragile]{Lab NumPy — Pregunta 5: ¿Qué predigo de \(M\) ensayos futuros? (predictiva de conteos)}
\temariobar{Teoría}{\textbf{Lab NumPy}}{Lab sklearn}{Entregable}

\begin{verbatim}
FIJA M (ensayos futuros) y R (numero de simulaciones)

PARA r = 1..R:
  muestrea theta_r ~ Beta(alpha_post, beta_post)
  muestrea S_r ~ Binomial(M, theta_r)

ESTIMA p(S | D) con {S_r} (histograma / frecuencias)
CHEQUEO: suma_s p_hat(S=s | D) ~= 1
\end{verbatim}
\begin{itemize}
  \item Definición: \(S=\sum_{j=1}^M y_j\) = número de éxitos futuros en \(M\) ensayos.
  \item Lógica Bayesiana: muestrear \(\theta\sim\mathrm{Beta}(\alpha',\beta')\) y luego \(S\sim\mathrm{Binomial}(M,\theta)\).
\end{itemize}
\end{frame}

\begin{frame}[fragile]{Lab NumPy — Pregunta 6: ¿Qué cambia si hago plug-in vs Bayes?}
\temariobar{Teoría}{\textbf{Lab NumPy}}{Lab sklearn}{Entregable}

\begin{verbatim}
ELIGE un estimador puntual para theta:
  si alpha_post > 1 y beta_post > 1:
    theta_map <- (alpha_post - 1) / (alpha_post + beta_post - 2)
  si no:
    theta_map <- alpha_post / (alpha_post + beta_post)

PREDICTIVA PLUG-IN:
  muestrea S_r_plugin ~ Binomial(M, theta_map)

COMPARA:
  compara media/varianza y colas de S_plugin vs S_bayes
\end{verbatim}
\begin{itemize}
  \item Definición: plug-in usa \(\theta=\hat\theta\) (punto); Bayes integra sobre \(\theta\).
  \item Nota: el modo (MAP) de una Beta es \(\dfrac{\alpha'-1}{\alpha'+\beta'-2}\) solo si \(\alpha',\beta'>1\).
  \item Qué comparar: varianza (dispersión) y colas (sobreconfianza).
\end{itemize}
\end{frame}

from sklearn.metrics import accuracy_score, precision_score, recall_score

taus = np.linspace(0.0, 1.0, 101)
rows = []
for tau in taus:
    yhat = (p_hat >= tau).astype(int)
    acc = accuracy_score(yte, yhat)
    prec = precision_score(yte, yhat, zero_division=0)
    rec  = recall_score(yte, yhat, zero_division=0)

best_tau_acc = rows[rows[:,1].argmax(), 0]
print("best tau (accuracy):", best_tau_acc)
