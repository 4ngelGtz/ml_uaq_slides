\begin{frame}{Decision theory: ingredientes básicos}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\begin{itemize}
  \item \key{Estados del mundo} \(h\in\mathcal{H}\): clase verdadera, valor real, etc.
  \item \key{Acciones} \(a\in\mathcal{A}\): predecir una clase, un número, dar/no dar un tratamiento, elegir un modelo, etc.
  \item \key{Pérdida/costo} \(\ell(h,a)\): qué tan “mala” es la acción \(a\) si la realidad es \(h\). \hfill \key{(5.1.1)}
  \item \key{Riesgo posterior} dado evidencia \(x\):
  \[
    \rho(a\mid x)=\mathbb{E}_{p(h\mid x)}[\ell(h,a)].
  \]
  \item \key{Regla de decisión Bayesiana}:
  \[
    \pi^*(x)=\arg\min_{a\in\mathcal{A}} \rho(a\mid x).
  \]
\item \key{Ejemplo mental}: médico que decide entre “no tratar” o “dar fármaco caro con efectos secundarios” a un paciente que puede (o no) tener COVID-19.
\end{itemize}
\end{frame}

\begin{frame}{Decision theory: ejemplo COVID-19 (tabla de pérdidas)}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\small
\begin{itemize}
  \item \key{Estados} \(h\in\mathcal{H}\): combinación de
  \begin{itemize}
    \item edad (joven vs viejo),
    \item estado de enfermedad (COVID-19 vs no COVID-19).
  \end{itemize}
  \item \key{Acciones} \(a\in\mathcal{A}\): \(\{\)\term{Nothing} (no tratar), \term{Drugs} (dar fármaco)\(\}\).
  \item Usamos \term{QALYs} (quality-adjusted life years) como unidad de costo/beneficio:
  \begin{itemize}\small
    \item joven: 60 QALYs esperados si se salva;
    \item viejo: 10 QALYs esperados si se salva;
    \item el fármaco “cuesta” 8 QALYs (dolor, efectos secundarios).
  \end{itemize}
  \item \key{QALY} (\term{quality-adjusted life year}) \(\approx\) “un año de vida en perfecta salud”; años con peor salud aportan menos de 1 QALY.
\end{itemize}

  \midrule
  No COVID-19, young  & 0  & 8 \\
  COVID-19, young     & 60 & 8 \\
  No COVID-19, old    & 0  & 8 \\

\end{tabular}

\end{frame}

\begin{frame}{Decision theory: cómo leer la tabla de pérdidas}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\small
\begin{itemize}
  \item La tabla no muestra los QALYs que la persona \emph{tiene}, sino cuántos QALYs \textbf{pierde} por la combinación estado/acción.
  \item Fila “No COVID-19, young”:
  \begin{itemize}\small
    \item \textbf{Nothing = 0}: el joven sano tiene muchos QALYs de vida potencial, pero la decisión de no tratar \emph{no le quita ninguno}.
    \item \textbf{Drugs = 8}: tratarlo innecesariamente le hace perder 8 QALYs (efectos secundarios, sufrimiento, secuelas).
  \end{itemize}
  \item Fila “COVID-19, young”:
  \begin{itemize}\small
    \item \textbf{Nothing = 60}: si no se trata y muere, se pierden los 60 QALYs que habría tenido.
    \item \textbf{Drugs = 8}: el tratamiento le salva la vida pero “cuesta” 8 QALYs de calidad.
  \end{itemize}
  \item Igual razonamiento para el paciente viejo con COVID-19 (10 vs 8 QALYs de pérdida).
\end{itemize}
\end{frame}

\begin{frame}{Riesgo esperado y política óptima (ejemplo COVID-19)}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\small
\begin{itemize}
  \item Dado lo que observamos \(x\) (p.\,ej. resultado del test y edad), el \key{riesgo} de una acción \(a\) es la pérdida esperada:
  \[
    \rho(a\mid x) = \mathbb{E}[\ell(h,a)\mid x] = \sum_{h\in\mathcal{H}} \ell(h,a)\,p(h\mid x).
  \]
  \item En el ejemplo, una acción es \(a=\textbf{Nothing}\) (\emph{no hacer nada}). Llamamos \key{cost-noop} al riesgo de esa acción:
  \[
    \text{cost-noop}(x) = \rho(\text{Nothing}\mid x).
  \]
  \item La \key{política óptima} elige, para cada \(x\), la acción con menor riesgo:
  \[
    \pi^*(x) = \arg\min_{a\in\mathcal{A}} \rho(a\mid x).
  \]
  \item Los \key{inputs} del ejemplo son: \texttt{test} (negativo/positivo), \texttt{age} (joven/viejo) y \(p(\text{COVID}\mid x)\). En la siguiente slide veremos cómo se calcula cost-noop; al final tendremos la tabla con la decisión óptima.
\end{itemize}
\end{frame}

\begin{frame}{Ejemplo COVID-19: cálculo de \texttt{cost-noop} (fórmula)}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\small
\begin{itemize}
  \item Por definición, el cost-noop es el riesgo de la acción ``no hacer nada'':
  \[
    \text{cost-noop}(x) = \mathbb{E}[\ell(h,\text{Nothing})\mid x]
    = \sum_h \ell(h,\text{Nothing})\,p(h\mid x).
  \]
  \item \textbf{Paciente joven}: de la tabla de pérdidas, \(\ell(\text{No COVID},\text{young},\text{Nothing})=0\) y \(\ell(\text{COVID},\text{young},\text{Nothing})=60\). Entonces
  \[
    \text{cost-noop} = 0\cdot p(\text{No COVID}\mid x) + 60\cdot p(\text{COVID}\mid x)
    = 60\,p(\text{COVID}\mid x).
  \]
  \item \textbf{Paciente mayor}: \(\ell(\text{No COVID},\text{old},\text{Nothing})=0\), \(\ell(\text{COVID},\text{old},\text{Nothing})=10\), luego
  \[
    \text{cost-noop} = 10\,p(\text{COVID}\mid x).
  \]
  \item Los valores de la tabla se obtienen sustituyendo \(p(\text{COVID}\mid x)\) por las probabilidades que da el test (siguiente slide).
\end{itemize}
\end{frame}

\begin{frame}{Ejemplo COVID-19: \texttt{cost-noop} (valores numéricos)}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\small
\begin{itemize}
  \item Con las probabilidades del test: \(p(\text{COVID}\mid \text{test}=0)\approx 0.014\), \(p(\text{COVID}\mid \text{test}=1)\approx 0.7955\).
  \item Sustituyendo en las fórmulas anteriores:
  \begin{itemize}\small
    \item \textbf{Test negativo}: joven \(60\times 0.014 = 0.84\); mayor \(10\times 0.014 = 0.14\).
    \item \textbf{Test positivo}: joven \(60\times 0.7955 = 47.73\); mayor \(10\times 0.7955 = 7.95\).
  \end{itemize}
\end{itemize}
\vspace{0.3em}
\centering
\begin{tabular}{cccccc}

  test & age & pr(covid) & cost-noop & cost-drugs & action \\

\end{itemize}
\end{frame}

\begin{frame}{De pérdida a utilidad: el mismo principio}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\begin{itemize}
  \item En muchos textos se usa \key{utilidad} \(U(h,a)\) en vez de pérdida \(\ell(h,a)\).
  \item Relación simple: \(U(h,a)=-\ell(h,a)\).
  \item \key{Máxima utilidad esperada}:
  \[
    \pi^*(x)=\arg\max_{a\in\mathcal{A}} \mathbb{E}_{p(h\mid x)}[U(h,a)].
  \]
  \item En ML solemos pensar en \key{minimizar pérdida promedio} (riesgo) \(\leftrightarrow\) \key{maximizar utilidad esperada}.
\end{itemize}
\end{frame}

\begin{frame}{Clasificación binaria con pérdida \(0\text{-}1\)}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\begin{itemize}
  \item Estados y acciones son las clases: \(\mathcal{H}=\mathcal{A}=\{0,1\}\). \hfill \key{(5.1.2.1)}
  \item \key{Pérdida \(0\text{-}1\)}:
  \[
    \ell_{0/1}(y^*,\hat{y}) = \mathbf{1}[y^*\neq \hat{y}].
  \]
  \item Riesgo posterior de predecir \(\hat{y}\):
  \[
    \rho(\hat{y}\mid x) = 1 - p(y^*=\hat{y}\mid x).
  \]
  \item \key{Regla óptima}: elegir la clase más probable (MAP):
  \[
    \hat{y}(x) = \arg\max_{y\in\{0,1\}} p(y\mid x).
  \]
\end{itemize}
\end{frame}

\begin{frame}{Costos asimétricos en binario: matriz de pérdida}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\small
\begin{itemize}
  \item En muchos problemas, \key{no todos los errores cuestan igual}.
  \item Matriz de pérdida general:
  \[
  \ell(y^*,\hat{y}) =
  \begin{pmatrix}
    \ell_{00} & \ell_{01} \\
    \ell_{10} & \ell_{11}
  \end{pmatrix},
  \]
  donde filas = verdad, columnas = predicción. \hfill \key{(5.1.2.2)}
  \item Ejemplo clásico: detección de enfermedad
  \begin{itemize}
    \item FP (decir “enfermo” cuando está sano) = costo moderado.
    \item FN (decir “sano” cuando está enfermo) = costo muy alto.
  \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Costos asimétricos \(\Rightarrow\) umbral distinto de 0.5}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\small
\begin{itemize}
  \item Sea \(p_1 = p(y^*=1\mid x)\), \(p_0 = 1-p_1\), donde \(y^*\) es la \key{etiqueta verdadera} (ground truth) y \(\hat{y}\) la predicción.
  \item Así, \(p_1\) = probabilidad de que la clase real sea 1 dado \(x\); \(p_0\) = probabilidad de que sea 0.
  \item Riesgo de predecir \(\hat{y}=1\):
  \[
    \rho(1\mid x) = \ell_{01}p_0 + \ell_{11}p_1.
  \]
  \item Riesgo de predecir \(\hat{y}=0\):
  \[
    \rho(0\mid x) = \ell_{00}p_0 + \ell_{10}p_1.
  \]
  \item Caso típico \(\ell_{00}=\ell_{11}=0\): predecir \(1\) si
  \[
    p_1 \ge \frac{\ell_{01}}{\ell_{01}+\ell_{10}}.
  \]
  \item Si un FN cuesta \(c\) veces más que un FP (\(\ell_{10}=c\,\ell_{01}\)):
  \[
    p_1 \ge \frac{1}{1+c}.
  \]
  \item \key{Ejemplo}: \(c=4\) \(\Rightarrow\) umbral \(1/5=0.2\) (somos mucho más “sensibles”).
\end{itemize}
\end{frame}

\begin{frame}{Costos asimétricos: de dónde sale el umbral}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\small
\begin{itemize}
  \item Con \(\ell_{00}=\ell_{11}=0\) (acertar no cuesta), el riesgo de predecir 1 es
    \(\rho(1\mid x)=\ell_{01}(1-p_1)\); el de predecir 0 es \(\rho(0\mid x)=\ell_{10}p_1\).
  \item Predecimos 1 cuando \(\rho(1\mid x)\le\rho(0\mid x)\), es decir
    \[
      \ell_{01}(1-p_1) \le \ell_{10}p_1
      \;\Longleftrightarrow\;
      \ell_{01} - \ell_{01}p_1 \le \ell_{10}p_1
      \;\Longleftrightarrow\;
      \ell_{01} \le p_1(\ell_{01}+\ell_{10})
      \;\Longleftrightarrow\;
      p_1 \ge \frac{\ell_{01}}{\ell_{01}+\ell_{10}}.
    \]
  \item Si \(\ell_{10}=c\,\ell_{01}\) (un FN cuesta \(c\) veces un FP):
    \[
      p_1 \ge \frac{1}{1+c}.
    \]
  \item \key{Interpretación}: \(c=4\) \(\Rightarrow\) umbral 0.2. Bajamos el umbral respecto a 0.5 para ser más sensibles: preferimos más FP que perder casos reales (FN).
\end{itemize}
\end{frame}

\begin{frame}{Costos asimétricos: umbral 0.5 vs umbral óptimo}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\small
\begin{itemize}
  \item Con \key{umbral 0.5} (costos simétricos): predecimos 1 solo si \(p_1\ge 0.5\).
  \item Con \key{umbral \(1/(1+c)\)} (FN cuesta \(c\) veces un FP): predecimos 1 si \(p_1\ge 1/(1+c)\).
  \item Si \(c>1\), entonces \(1/(1+c)<0.5\): el umbral óptimo es \emph{menor} que 0.5.
  \item \key{Ejemplo} \(c=4\): umbral \(0.2\). Un modelo que da \(p_1=0.3\):
  \begin{itemize}\small
    \item Con 0.5: predeciríamos 0 (``sano''); con 0.2: predecimos 1 (``tratar'').
    \item Priorizamos no perder enfermos (menos FN), aunque haya más falsas alarmas (más FP).
  \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Costos asimétricos: ejemplo numérico completo}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\small
\begin{itemize}
  \item Supongamos la siguiente matriz de pérdida para un clasificador médico:
  \[
  \ell(y^*,\hat{y}) =
  \begin{pmatrix}
    0 & 1 \\
    5 & 0
  \end{pmatrix},
  \]
  donde \(y^*=0\) = sano, \(y^*=1\) = enfermo.
  \item Entonces un FN (enfermo etiquetado como sano) cuesta 5 veces un FP \(\Rightarrow c=5\).
  \item El \key{umbral óptimo} según decision theory es
  \[
    p_1^\star = \frac{1}{1+c} = \frac{1}{6}\approx 0.17.
  \]
  \item \key{Interpretación práctica}:
  \begin{itemize}
    \item Si \(p(y=1\mid x)=0.20\): bajo \(0.5\) dirías “sano”; bajo \(0.17\) ya vale la pena tratar.
    \item Si \(p(y=1\mid x)=0.05\): incluso con costos asimétricos, sigue por debajo del umbral \(0.17\).
  \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Matriz de confusión: conectar decisiones con conteos}

\begin{frame}{Derivar métricas desde la matriz de confusión}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\small
\begin{itemize}
  \item Normalizando por filas (condicionado en la verdad): \hfill \key{(5.1.3.1, 5.1.3.2)}
  \begin{itemize}
    \item \key{TPR / recall / sensitivity}: \( \mathrm{TPR} = \dfrac{TP}{TP + FN} \).
    \item \key{TNR / specificity}: \( \mathrm{TNR} = \dfrac{TN}{TN + FP} \).
    \item \key{FPR}: \( \mathrm{FPR} = 1-\mathrm{TNR} = \dfrac{FP}{FP + TN} \).
  \end{itemize}
  \item Normalizando por columnas (condicionado en la predicción):
  \begin{itemize}
    \item \key{Precision / PPV}: \( \mathrm{Prec} = \dfrac{TP}{TP + FP} \).
    \item \key{NPV}: \( \mathrm{NPV} = \dfrac{TN}{TN + FN} \).
  \end{itemize}
  \item \key{Idea}: todas las métricas que usamos (accuracy, recall, precision, etc.) son distintas “vistas” de la misma tabla.
\end{itemize}
\end{frame}

\begin{frame}{Matriz de confusión: ejemplo numérico}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\small
\begin{columns}[T,onlytextwidth]
\column{0.48\textwidth}
\centering
\begin{tabular}{c|cc}
      & \(\hat{y}=0\) & \(\hat{y}=1\) \\\hline
  \(y=0\) & 50 & 10 \\
  \(y=1\) & 5  & 35
\end{tabular}

\vspace{0.5em}
\begin{itemize}\small
  \item Total negativos verdaderos \(N=60\), positivos verdaderos \(P=40\).
  \item Accuracy \(= (50+35)/100 = 0.85\).
\end{itemize}
\column{0.52\textwidth}
\begin{itemize}\small
  \item \key{Recall / TPR}:
  \[
    \mathrm{TPR} = \frac{TP}{TP+FN} = \frac{35}{35+5}=0.875.
  \]
  \item \key{FPR}:
  \[
    \mathrm{FPR} = \frac{FP}{FP+TN} = \frac{10}{10+50}\approx 0.167.
  \]
  \item \key{Precision}:
  \[
    \mathrm{Prec} = \frac{TP}{TP+FP} = \frac{35}{35+10}\approx 0.78.
  \]
  \item \key{NPV}:
  \[
    \mathrm{NPV} = \frac{TN}{TN+FN} = \frac{50}{50+5}\approx 0.91.
  \]
\end{itemize}

\begin{frame}{Curva ROC: barrer el umbral}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\small
\begin{itemize}
  \item Si el modelo produce un \key{score} \(s(x)\) (p.\,ej. \(p(y=1\mid x)\)), definimos \(\hat{y}_\tau(x)=\mathbf{1}[s(x)\ge \tau]\).
  \item Para cada \(\tau\) calculamos \(\mathrm{TPR}_\tau\) y \(\mathrm{FPR}_\tau\). \hfill \key{(5.1.3)}
  \item La \key{curva ROC} grafica \(\mathrm{TPR}_\tau\) vs \(\mathrm{FPR}_\tau\) al variar \(\tau\in[0,1]\).
  \item \key{Extremos}:
  \begin{itemize}
    \item \(\tau\to 0\): todo es positivo (\(\mathrm{TPR}\approx 1\), \(\mathrm{FPR}\approx 1\)).
    \item \(\tau\to 1\): nadie es positivo (\(\mathrm{TPR}\approx 0\), \(\mathrm{FPR}\approx 0\)).
  \end{itemize}
\end{itemize}

\vspace{0.2em}
\centering
\includegraphics[width=0.85\linewidth,height=0.42\textheight,keepaspectratio]{pr_curve.png}
\end{frame}

\begin{frame}{AUC, EER y lectura rápida de ROC}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\small
\begin{itemize}
  \item \key{AUC} (Area Under the ROC Curve):
  \begin{itemize}
    \item \(\mathrm{AUC}=1\): clasificador perfecto (curva pegada a la esquina superior izquierda).
    \item \(\mathrm{AUC}\approx 0.5\): como lanzar una moneda (diagonal).
  \end{itemize}
  \item \key{Equal Error Rate (EER)}:
  \begin{itemize}
    \item Punto donde \(\mathrm{FPR} = \mathrm{FNR} = 1-\mathrm{TPR}\). \hfill \key{(5.1.3.2)}
    \item Se visualiza como intersección de la ROC con la diagonal descendente.
  \end{itemize}
  \item \key{Uso práctico}: ROC es útil cuando el costo de FP/FN no está fijado, y queremos comparar modelos “globalmente”.
\end{itemize}
\end{frame}

\begin{frame}{Curva Precision–Recall (PR): cuando los positivos son raros}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\small
\begin{itemize}
  \item Recordemos:
  \[
    \mathrm{Prec}_\tau = \frac{TP_\tau}{TP_\tau+FP_\tau},\qquad
    \mathrm{Rec}_\tau = \frac{TP_\tau}{TP_\tau+FN_\tau}. \qquad \key{(5.1.4.1)}
  \]
  \item La \key{curva PR} grafica \(\mathrm{Prec}_\tau\) vs \(\mathrm{Rec}_\tau\) al variar \(\tau\).
  \item Es especialmente informativa cuando hay \key{fuerte desbalance} (\(P\ll N\)), donde ROC puede verse “demasiado bonita”.
  \item \key{PR-AUC / AP}: área (o average precision) bajo la curva PR. \hfill \key{(5.1.4.2)}
\end{itemize}

\centering
\includegraphics[width=0.85\linewidth,height=0.42\textheight,keepaspectratio]{pr_curve.png}
\end{frame}

\begin{frame}{F-scores: combinar precisión y recall en un solo número}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\small
\begin{itemize}
  \item Para un umbral fijo (\(\mathrm{Prec},\mathrm{Rec}\)), el \key{F\(_\beta\)} se define como: \hfill \key{(5.1.4.3)}
  \[
    F_\beta = \frac{(1+\beta^2)\,\mathrm{Prec}\,\mathrm{Rec}}{\beta^2\,\mathrm{Prec} + \mathrm{Rec}}.
  \]
  \item Casos especiales:
  \begin{itemize}
    \item \(F_1\): media armónica de precisión y recall (pesa ambos por igual).
    \item \(\beta>1\): da más peso a \term{recall} (importa más cubrir positivos).
    \item \(\beta<1\): da más peso a \term{precision} (importa más evitar FP).
  \end{itemize}
  \item \key{Interpretación}: útil para seleccionar el umbral cuando tenemos una preferencia clara por precision o recall.
\end{itemize}
\end{frame}

\begin{frame}{Desbalance de clases: ROC vs PR}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\small
\begin{itemize}
  \item ROC usa tasas \(\mathrm{TPR}\) y \(\mathrm{FPR}\) que \key{no cambian} si escalamos P y N manteniendo la proporción. \hfill \key{(5.1.3.3)}
  \item En problemas muy desbalanceados, ROC puede verse buena aunque haya muchos FP absolutos.
  \item PR (y F-score) sí son \key{sensibles} a la prevalencia:
  \[
    \mathrm{Prec} = \frac{\mathrm{TPR}}{\mathrm{TPR} + \frac{1}{r}\mathrm{FPR}},\quad
    r=\frac{P}{N}. \qquad \key{(5.1.4.4)}
  \]
  \item \key{Regla práctica}:
  \begin{itemize}
    \item Clases balanceadas \(\rightarrow\) ROC-AUC es razonable.
    \item Clases raras \(\rightarrow\) mira PR-AUC / AP y F-scores.
  \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Validación: train / validation / test y leakage}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\small
\begin{itemize}
  \item Separar los datos en:
  \begin{itemize}
    \item \key{Train}: ajustar parámetros \(\theta\).
    \item \key{Validation}: escoger hiperparámetros (p.\,ej. \(\lambda\), arquitectura, umbral).
    \item \key{Test}: estimar \(L(\theta^*;p^*)\) una sola vez. \hfill \key{(1.2.3, 4.5.4)}
  \end{itemize}
  \item \key{Leakage}: usar test (directa o indirectamente) para tomar decisiones de modelo/hiperparámetros.
  \item \key{Consecuencia}: subestimar el error real; el modelo parece mejor de lo que realmente es.
\end{itemize}
\end{frame}

\begin{frame}{Cross-validation (CV): usar mejor los datos}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\small
\begin{itemize}
  \item Problema: con pocos datos, separar un validation set fijo puede ser caro.
  \item \key{K-fold CV} (\(K\)-pliegues): \hfill \key{(4.5.5, 5.4.3)}
  \begin{itemize}
    \item Partimos train en \(K\) folds \(D_1,\dots,D_K\).
    \item Para cada \(k\): entrenamos en \(D_{-k}\) y evaluamos en \(D_k\).
    \item Promediamos la métrica (MSE, AUC, F1, etc.) sobre los \(K\) folds.
  \end{itemize}
  \item \key{Uso típico}: escoger hiperparámetros (p.\,ej. \(\lambda\) de regularización) y comparar modelos.
\end{itemize}

\begin{frame}{Cross-validation (CV): ejemplo numérico sencillo}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\small
\begin{itemize}
  \item Supongamos \(K=3\) folds y medimos \key{accuracy} para un modelo:
\end{itemize}
\vspace{-0.4em}
\[
\begin{array}{c|c}
  \text{Fold} & \text{Accuracy} \\\hline
  1 & 0.82 \\
  2 & 0.79 \\
  3 & 0.87
\end{array}
\]
\vspace{-0.4em}
\begin{itemize}\small
  \item \key{Promedio}: \(\bar{a} = (0.82+0.79+0.87)/3 \approx 0.83\).
  \item \key{Desviación estándar} (muestral):
  \[
    s \approx \sqrt{\frac{(0.82-\bar{a})^2+(0.79-\bar{a})^2+(0.87-\bar{a})^2}{3-1}} \approx 0.04.
  \]
  \item Reporte típico: \(\text{accuracy CV} \approx 0.83 \pm 0.04\).
  \item \key{Lectura}: el modelo parece razonablemente estable, pero hay variabilidad entre folds (dependiente del split).
\end{itemize}
\end{frame}
