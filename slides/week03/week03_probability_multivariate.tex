\section[Teoría]{Teoría\label{sec:teoria}}

\begin{frame}{Punto de partida: vector aleatorio, media y covarianza}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\begin{itemize}
  \item En datos reales (p.\,ej. Iris), cada fila es un vector \(x\in\mathbb{R}^D\) (aquí \(D=4\)).
  \item Modelamos \(X\) como vector aleatorio: \(X\in\mathbb{R}^D\).
  \item \key{Media}: \(\mu=\E[X]\) (valor “típico” por variable).
  \item \key{Covarianza}: \(\Sigma=\E[(X-\mu)(X-\mu)^\top]\) (escala y dependencia lineal).
  \item \key{Entrada}: \(\Sigma_{ij}=\mathrm{Cov}(X_i,X_j)\) indica si dos variables tienden a subir/bajar juntas.
\end{itemize}
\end{frame}

\begin{frame}{Construcción en práctica: \(\hat{\Sigma}\) y \(\hat{R}\) (pasos)}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\small
\begin{enumerate}
  \item \key{Paso 1 — datos:} Tenemos \(n\) observaciones \(x_1,\dots,x_n\) (filas). Puede ser Iris (4 variables) o un ejemplo 3D.
  \item \key{Paso 2 — covarianza muestral:} Estimamos \(\hat{\Sigma}_{ij}\) como el promedio centrado de productos de desviaciones:
  \[
  \hat{\Sigma}_{ij}=\frac{1}{n-1}\sum_{k=1}^{n}(x_{ki}-\bar{x}_i)(x_{kj}-\bar{x}_j),
  \quad \bar{x}_i=\frac{1}{n}\sum_{k=1}^{n}x_{ki}.
  \]
  Diagonal = varianza muestral; fuera de diagonal = covarianza (nota: \term{unidades mezcladas}).
  \item \key{Paso 3 — correlación:} Estandarizamos a escala \([-1,1]\):
  \[
  \hat{R}_{ij}=\frac{\hat{\Sigma}_{ij}}{\sqrt{\hat{\Sigma}_{ii}\hat{\Sigma}_{jj}}}.
  \]
  \item \key{Lectura rápida:} usa \(\hat{R}\) para comparar dependencias; usa \(\hat{\Sigma}\) para cálculos (geometría y condicionamiento).
\end{enumerate}
\end{frame}

\begin{frame}{Alerta 1: \(\rho=0\) no implica independencia}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\begin{columns}[T,onlytextwidth]
\column{0.55\textwidth}
\begin{itemize}
  \item \key{Mensaje}: correlación mide \term{dependencia lineal}. Puede haber relación no lineal con \(\rho\approx 0\).
  \item \key{Hecho}: \( \mathrm{Cov}(X,Y)=0 \) \(\not\Rightarrow\) \(X\perp Y\) (salvo casos especiales como gaussianas).
  \item \key{Traducción práctica}: “no hay correlación” \(\neq\) “no hay relación”.
\end{itemize}
\column{0.45\textwidth}
\centering
\includegraphics[width=\linewidth]{week03_uncorrelated_dependent.png}
\end{columns}
\end{frame}

\begin{frame}{Alerta 2: correlación no implica causalidad}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\begin{itemize}
  \item Una correlación puede venir de \key{confusores}, \key{sesgo de selección} o \key{causalidad invertida}.
  \item \key{Pregunta guía}: ¿qué variable podría estar afectando a ambas? ¿qué estoy dejando fuera?
\end{itemize}
\end{frame}

\begin{frame}{Paradoja de Simpson: cuando el agregado engaña}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\begin{itemize}
  \item \key{Definición}: una tendencia en datos \key{agregados} cambia (o se invierte) al \key{estratificar} por una tercera variable.
  \item \key{Ejemplo (Berkeley, 1973)}: al controlar por departamento, la interpretación cambia.
  \begin{itemize}
    \item \key{Agregado}: parece haber sesgo.
    \item \key{Estratificado}: aparece el rol del confusor (departamento).
  \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{MVN: un baseline probabilístico para vectores}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\begin{itemize}
  \item Idea: aproximar una nube de datos con una sola \term{elipse} (2D) / \term{elipsoide} (D).
  \item Notación: \(X\sim\mathcal{N}(\mu,\Sigma)\).
  \item \(\mu\): centro; \(\Sigma\): escala + orientación (forma).
  \item Esto nos da dos cosas: \key{geometría} (elipses) y \key{cálculo cerrado} (condicionales).
\end{itemize}
\end{frame}

\begin{frame}{Geometría de \(\Sigma\): cómo “lee” la MVN la nube de datos}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\begin{columns}[T,onlytextwidth]
\column{0.52\textwidth}
\begin{itemize}
  \item En una MVN, las curvas de igual densidad son \term{elipses}/\term{elipsoides}.
  \item \key{Autovectores}: direcciones principales (ejes) de la nube.
  \item \key{Autovalores}: dispersión en cada dirección (eje \(\propto\sqrt{\lambda}\)).
  \item \key{Lectura práctica}: \(\Sigma\) codifica \term{escala} y \term{correlación} en una sola matriz.
\end{itemize}
\column{0.48\textwidth}
\centering
\includegraphics[width=\linewidth]{week03_covariance_geometry.png}
\end{columns}
\end{frame}

\begin{frame}{Distancia de Mahalanobis: qué es y para qué sirve}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\begin{columns}[T,onlytextwidth]
\column{0.55\textwidth}
\begin{itemize}
  \item \key{Definición:}
  \[
    d_M(x,\mu)=\sqrt{(x-\mu)^\top\Sigma^{-1}(x-\mu)}.
  \]
  \item \key{Intuición:} mide “lejanía” en unidades que respetan \term{escala} y \term{correlación}.
  \item \key{Sirve para:}
  \begin{itemize}
    \item detectar \term{atípicos} bajo una MVN (outliers),
    \item entender el \term{log-likelihood} gaussiano (el término cuadrático),
    \item conectar con GMM: cada componente usa su \(\Sigma_k\) para medir cercanía.
  \end{itemize}
  \item \key{Puente:} la geometría que define \(d_M\) es la misma que usamos al \key{condicionar} (siguiente bloque).
\end{itemize}
\column{0.45\textwidth}
\centering
\includegraphics[width=\linewidth]{week03_mahalanobis.png}
\end{columns}
\end{frame}

\begin{frame}{De “medir rareza” a “inferir faltantes”}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\begin{itemize}
  \item Con \(\mu\) y \(\Sigma\) ya podemos:
  \begin{itemize}
    \item evaluar qué tan plausible es un punto (vía \(d_M\)),
    \item y también \key{inferir} partes faltantes de un vector.
  \end{itemize}
  \item La herramienta clave es que en la MVN, \key{marginales} y \key{condicionales} siguen siendo gaussianas.
\end{itemize}
\end{frame}

\begin{frame}{Marginales y condicionales en una MVN (la pieza que habilita todo)}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\begin{itemize}
  \item \key{Marginal}: si te quedas con algunas coordenadas, sigues en el mundo gaussiano.
  \item \key{Condicional}: si conoces parte del vector, la distribución del resto también es gaussiana.
  \item \key{Consecuencia}: podemos calcular \(\E[X_{\text{faltante}}\mid X_{\text{observado}}]\) en forma cerrada \(\Rightarrow\) imputación.
\end{itemize}
\end{frame}

\begin{frame}{Condicionamiento por bloques: la receta (MVN)}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\begin{itemize}
  \item \key{Notación}: \(a\) = variables \term{observadas}; \(b\) = variables \term{faltantes} (o las que queremos predecir).
  \item Partición:
  \[
    x=\begin{pmatrix}x_a\\x_b\end{pmatrix},\quad
    \mu=\begin{pmatrix}\mu_a\\\mu_b\end{pmatrix},\quad
    \Sigma=\begin{pmatrix}\Sigma_{aa}&\Sigma_{ab}\\ \Sigma_{ba}&\Sigma_{bb}\end{pmatrix}.
  \]
  \item Resultado: \(X_b\mid X_a=a\sim \mathcal{N}(\mu_{b\mid a},\Sigma_{b\mid a})\).
  \item Fórmulas:
  \[
    \mu_{b\mid a}=\mu_b+\Sigma_{ba}\Sigma_{aa}^{-1}(a-\mu_a),
    \qquad
    \Sigma_{b\mid a}=\Sigma_{bb}-\Sigma_{ba}\Sigma_{aa}^{-1}\Sigma_{ab}.
  \]
  \item Nota: \(\Sigma_{b\mid a}\) no depende de \(a\). Para imputación “simple”, usamos la media \(\mu_{b\mid a}\).
\end{itemize}
\end{frame}

\begin{frame}{Condicionamiento por bloques: ejemplo numérico (D=5)}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\small
\key{Ejemplo:} observamos 3 variables (\(a\)) y faltan 2 (\(b\)). Así “se ven” \(\mu\) y \(\Sigma\).
\begin{itemize}\small
  \item Vector de medias: \(\mu=\begin{pmatrix}2.1\\1.8\\3.0\\4.2\\5.5\end{pmatrix}\) (shape \((5,)\)).
  \item Matriz de covarianza \(\Sigma\) (shape \((5\times 5)\)):
  \[
  \Sigma=\begin{pmatrix}
    1.2 & 0.5 & 0.3 & 0.4 & 0.2\\
    0.5 & 0.9 & 0.2 & 0.3 & 0.1\\
    0.3 & 0.2 & 1.5 & 0.6 & 0.4\\
    0.4 & 0.3 & 0.6 & 2.0 & 0.8\\
    0.2 & 0.1 & 0.4 & 0.8 & 1.8
  \end{pmatrix}
  \]
  \item Partición: \(a\) = índices \([0,1,2]\) (observadas), \(b\) = índices \([3,4]\) (faltantes).
\end{itemize}
\end{frame}

\begin{frame}{Condicionamiento por bloques: ejemplo numérico (D=5) — subbloques}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\small
\begin{itemize}\small
  \item \key{Partir \(\mu\)} en observado/faltante:
  \[
  \mu_a=\begin{pmatrix}2.1\\1.8\\3.0\end{pmatrix}\quad\text{(shape }(3,)\text{)},\qquad
  \mu_b=\begin{pmatrix}4.2\\5.5\end{pmatrix}\quad\text{(shape }(2,)\text{)}.
  \]
  \item \key{Partir \(\Sigma\)} en 4 bloques (observado-observado, observado-faltante, etc.):
  \[
  \Sigma_{aa}=\begin{pmatrix}1.2&0.5&0.3\\0.5&0.9&0.2\\0.3&0.2&1.5\end{pmatrix}\quad\text{(shape }(3\times 3)\text{)},
  \]
  \[
  \Sigma_{ab}=\begin{pmatrix}0.4&0.2\\0.3&0.1\\0.6&0.4\end{pmatrix}\quad\text{(shape }(3\times 2)\text{)},
  \qquad
  \Sigma_{ba}=\Sigma_{ab}^\top,\quad
  \Sigma_{bb}=\begin{pmatrix}2.0&0.8\\0.8&1.8\end{pmatrix}\quad\text{(shape }(2\times 2)\text{)}.
  \]
  \item \key{Lectura}: \(\Sigma_{aa}\) es lo que invertimos; \(\Sigma_{ba}\) “transporta” información de observado \(\rightarrow\) faltante.
  \item Para imputar: \(\mu_{b\mid a}=\mu_b+\Sigma_{ba}\Sigma_{aa}^{-1}(x_a-\mu_a)\) (shape \((2,)\)).
\end{itemize}
\end{frame}

\begin{frame}{Imputación (Iris) = condicionamiento + evaluación honesta}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\begin{columns}[T,onlytextwidth]
\column{0.52\textwidth}
\begin{itemize}
  \item \key{Idea}: tratar “faltante” como \(X_m\) y lo observado como \(X_o\).
  \item \key{Regla}: imputación determinista \(x_m \leftarrow \E[X_m\mid X_o=x_o]\) (media condicional).
  \item \key{Experimento}: ocultamos ancho del pétalo en 20\,\% de filas; estimamos \((\mu,\Sigma)\) con las filas observadas; imputamos y comparamos.
  \item \key{Gráfica}: cada punto azul = una celda ocultada (verdadero vs imputado). Diagonal \(y=x\) = ideal.
  \item \key{Métrica}: MSE \textbf{solo} donde ocultaste; compara contra baseline (media marginal por columna).
\end{itemize}

\begin{frame}{¿Por qué no basta una sola MVN? (puente a mixtures)}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\begin{itemize}
  \item Una MVN = una sola “nube elíptica”.
  \item En Iris, distintas especies sugieren \key{más de una nube} (multimodalidad).
  \item Solución: \key{mezcla} = suma ponderada de varias MVNs.
\end{itemize}
\end{frame}

\begin{frame}{GMM: varias elipses para explicar una nube compleja}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\begin{itemize}
  \item Modelo:
  \[
    p(x)=\sum_{k=1}^K \pi_k\,\mathcal{N}(x\mid \mu_k,\Sigma_k),
    \qquad \sum_k\pi_k=1,\; \pi_k\ge 0.
  \]
  \item En 2D, cada componente induce una elipse; el modelo suma “nubes”.
  \item Interpretación: \(\pi_k\) (peso), \(\mu_k\) (centro), \(\Sigma_k\) (forma).
  \item Conexión: “cercanía” a cada componente se relaciona con Mahalanobis y su log-likelihood.
\end{itemize}
\end{frame}

\begin{frame}{Cierre: qué te llevas de Week03}
\temariobar{\textbf{Teoría}}{Lab NumPy}{Lab sklearn}{Entregable}
\begin{itemize}
  \item \(\hat{\Sigma}\) conserva escala; \(\hat{R}\) estandariza para comparar dependencias.
  \item Interpretación cuidadosa: \(\rho=0\not\Rightarrow\) independencia; correlación \(\not\Rightarrow\) causalidad; cuidado con Simpson.
  \item MVN + \(\Sigma\) da geometría; Mahalanobis mide “rareza” bajo esa geometría.
  \item Condicionamiento MVN es cerrado \(\Rightarrow\) imputación como \(\E[X_m\mid X_o]\).
  \item Si hay varias nubes, GMM (mezcla de elipses) es una extensión natural.
\end{itemize}
\end{frame}

\begin{frame}{Lab NumPy — notación y shapes (traducción Teoría \(\rightarrow\) código)}
\temariobar{Teoría}{\textbf{Lab NumPy}}{Lab sklearn}{Entregable}
\small
\begin{itemize}
  \item \key{\texttt{mu (D,)}}: vector de medias; \texttt{mu[i]} = media de la variable \(i\).
  \item \key{\texttt{Sigma (D,D)}}: covarianza; \texttt{Sigma[i,j]} = cov(\(X_i,X_j\)).
  \item \key{\texttt{idx\_a / idx\_b}}: índices de observado/faltante (en Iris, \(D=4\) y puedes ocultar “petal width”). 
  \item \key{\texttt{x\_a}}: valores observados de una fila; es el “\(a\)” numérico de esa observación.
  \item Fórmula por fila (media condicional): \(\mu_{b\mid a}=\mu_b+\Sigma_{ba}\Sigma_{aa}^{-1}(x_a-\mu_a)\).
\end{itemize}
\end{frame}

\begin{frame}[fragile]{Lab NumPy — Pseudocódigo 1: partir en bloques}
\temariobar{Teoría}{\textbf{Lab NumPy}}{Lab sklearn}{Entregable}
\begin{verbatim}
# INPUT:
#   mu:    shape (D,)    # vector de medias
#   Sigma: shape (D,D)   # matriz de covarianza
#   idx_a: índices observados
#   idx_b: índices faltantes

mu_a = mu[idx_a]                      # (Da,)
mu_b = mu[idx_b]                      # (Db,)
S_aa = Sigma[np.ix_(idx_a, idx_a)]    # (Da,Da)
S_ab = Sigma[np.ix_(idx_a, idx_b)]    # (Da,Db)
S_ba = Sigma[np.ix_(idx_b, idx_a)]    # (Db,Da)
S_bb = Sigma[np.ix_(idx_b, idx_b)]    # (Db,Db)

# CHEQUEO: S_aa debe ser invertible (si no, regulariza: S_aa += eps*I)
\end{verbatim}
\end{frame}

\begin{frame}[fragile]{Lab NumPy — Pseudocódigo 2: media condicional \(\rightarrow\) imputación}
\temariobar{Teoría}{\textbf{Lab NumPy}}{Lab sklearn}{Entregable}
\begin{verbatim}
# Precompute una vez (fuera del loop por filas):
S_aa_inv = inv(S_aa)   # costoso -> hazlo una sola vez

# Para cada fila i con faltantes en idx_b:
x_a = X[i, idx_a]      # valores observados de la fila (Da,)

# Fórmula MVN:
# mu_{b|a} = mu_b + S_ba @ inv(S_aa) @ (x_a - mu_a)
mu_b_given_a = mu_b + S_ba @ S_aa_inv @ (x_a - mu_a)  # (Db,)

# Imputación determinista (media condicional):
X_imputed[i, idx_b] = mu_b_given_a

# Opcional (imputación estocástica):
# S_{b|a} = S_bb - S_ba @ inv(S_aa) @ S_ab
\end{verbatim}
\end{frame}

\begin{frame}[fragile]{Lab NumPy — Pseudocódigo 3: evaluación correcta (MSE)}
\temariobar{Teoría}{\textbf{Lab NumPy}}{Lab sklearn}{Entregable}
\begin{verbatim}
# 1) Guarda el dato completo
X_true = X.copy()                    # (N,D)

# 2) Genera máscara de celdas a ocultar (True = oculto)
mask = make_mask(...)                # (N,D) boolean

# 3) Oculta y luego imputa
X_missing = X_true.copy()
X_missing[mask] = NaN
X_imputed = impute(X_missing)        # tu función del pseudocódigo 2

# 4) Evalúa solo donde ocultaste (para no "hacer trampa")
mse = mean( (X_true[mask] - X_imputed[mask])**2 )

# Baseline: imputar por media marginal (media de columna)
\end{verbatim}
\end{frame}
